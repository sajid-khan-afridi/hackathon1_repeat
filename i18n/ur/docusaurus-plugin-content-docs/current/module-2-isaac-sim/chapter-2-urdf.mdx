# باب 2: URDF روبوٹ ماڈلز

## تعارف

<PersonalizedSection level="beginner" rosFamiliarity="novice">
<GlossaryTooltip termId="urdf">URDF</GlossaryTooltip> (Unified Robot Description Format) ایک XML فارمیٹ ہے جو روبوٹ کی ساخت کو بیان کرتا ہے۔ اسے روبوٹ کے "نقشے" کے طور پر سمجھیں جو بتاتا ہے کہ اس کے حصے کیسے جڑے ہیں۔
</PersonalizedSection>

<PersonalizedSection level="advanced" rosFamiliarity="expert">
URDF روبوٹ کی kinematic اور dynamic خصوصیات کو define کرتا ہے، بشمول links، joints، collision geometry، visual meshes، اور inertial properties۔ Isaac Sim URDF کو USD format میں convert کرتا ہے physics simulation کے لیے۔
</PersonalizedSection>

## 2.1 URDF کی بنیادیں

### Link Element

Link روبوٹ کا rigid body ہے:

```xml
<link name="base_link">
  <!-- Visual representation -->
  <visual>
    <geometry>
      <box size="0.5 0.3 0.2"/>
    </geometry>
    <material name="blue">
      <color rgba="0 0 1 1"/>
    </material>
  </visual>

  <!-- Collision geometry -->
  <collision>
    <geometry>
      <box size="0.5 0.3 0.2"/>
    </geometry>
  </collision>

  <!-- Inertial properties -->
  <inertial>
    <mass value="10.0"/>
    <inertia ixx="0.1" ixy="0" ixz="0"
             iyy="0.1" iyz="0" izz="0.1"/>
  </inertial>
</link>
```

### Joint Element

Joint دو links کو جوڑتا ہے:

```xml
<!-- Revolute joint (rotation) -->
<joint name="wheel_joint" type="revolute">
  <parent link="base_link"/>
  <child link="wheel_link"/>
  <origin xyz="0.2 0 -0.1" rpy="0 0 0"/>
  <axis xyz="0 1 0"/>
  <limit lower="-3.14" upper="3.14" velocity="10" effort="100"/>
</joint>

<!-- Prismatic joint (translation) -->
<joint name="slider_joint" type="prismatic">
  <parent link="base_link"/>
  <child link="slider_link"/>
  <axis xyz="1 0 0"/>
  <limit lower="0" upper="0.5" velocity="1" effort="50"/>
</joint>

<!-- Fixed joint (no movement) -->
<joint name="sensor_mount" type="fixed">
  <parent link="base_link"/>
  <child link="sensor_link"/>
  <origin xyz="0 0 0.15"/>
</joint>
```

## 2.2 مکمل روبوٹ مثال

### سادہ موبائل روبوٹ

```xml
<?xml version="1.0"?>
<robot name="simple_robot">
  <!-- Base link -->
  <link name="base_link">
    <visual>
      <geometry>
        <box size="0.4 0.3 0.1"/>
      </geometry>
      <material name="grey">
        <color rgba="0.5 0.5 0.5 1"/>
      </material>
    </visual>
    <collision>
      <geometry>
        <box size="0.4 0.3 0.1"/>
      </geometry>
    </collision>
    <inertial>
      <mass value="5.0"/>
      <inertia ixx="0.05" ixy="0" ixz="0"
               iyy="0.08" iyz="0" izz="0.1"/>
    </inertial>
  </link>

  <!-- Left wheel -->
  <link name="left_wheel">
    <visual>
      <geometry>
        <cylinder radius="0.05" length="0.02"/>
      </geometry>
      <material name="black">
        <color rgba="0.1 0.1 0.1 1"/>
      </material>
    </visual>
    <collision>
      <geometry>
        <cylinder radius="0.05" length="0.02"/>
      </geometry>
    </collision>
    <inertial>
      <mass value="0.5"/>
      <inertia ixx="0.001" ixy="0" ixz="0"
               iyy="0.001" iyz="0" izz="0.001"/>
    </inertial>
  </link>

  <joint name="left_wheel_joint" type="continuous">
    <parent link="base_link"/>
    <child link="left_wheel"/>
    <origin xyz="0 0.16 -0.05" rpy="-1.5708 0 0"/>
    <axis xyz="0 0 1"/>
  </joint>

  <!-- Right wheel (similar structure) -->
  <link name="right_wheel">
    <visual>
      <geometry>
        <cylinder radius="0.05" length="0.02"/>
      </geometry>
      <material name="black"/>
    </visual>
    <collision>
      <geometry>
        <cylinder radius="0.05" length="0.02"/>
      </geometry>
    </collision>
    <inertial>
      <mass value="0.5"/>
      <inertia ixx="0.001" ixy="0" ixz="0"
               iyy="0.001" iyz="0" izz="0.001"/>
    </inertial>
  </link>

  <joint name="right_wheel_joint" type="continuous">
    <parent link="base_link"/>
    <child link="right_wheel"/>
    <origin xyz="0 -0.16 -0.05" rpy="-1.5708 0 0"/>
    <axis xyz="0 0 1"/>
  </joint>
</robot>
```

## 2.3 Isaac Sim میں لوڈ کرنا

### Python Script

```python
# test: true
# Isaac Sim میں URDF لوڈ کرنا
from omni.isaac.kit import SimulationApp

simulation_app = SimulationApp({"headless": False})

from omni.isaac.core import World
from omni.isaac.core.utils.stage import add_reference_to_stage
from omni.isaac.urdf import _urdf as omni_urdf

# URDF لوڈ کریں
urdf_interface = omni_urdf.acquire_urdf_interface()

import_config = omni_urdf.ImportConfig()
import_config.merge_fixed_joints = False
import_config.fix_base = True
import_config.make_default_prim = True
import_config.self_collision = False

# URDF سے USD بنائیں
urdf_path = "/path/to/robot.urdf"
dest_path = "/World/Robot"

urdf_interface.parse_urdf(urdf_path, import_config)
urdf_interface.import_robot(
    urdf_path,
    dest_path,
    import_config
)

# سیمولیشن چلائیں
world = World()
world.reset()

while simulation_app.is_running():
    world.step(render=True)

simulation_app.close()
```

## 2.4 Xacro کا استعمال

Xacro URDF کے لیے macros فراہم کرتا ہے:

```xml
<?xml version="1.0"?>
<robot xmlns:xacro="http://www.ros.org/wiki/xacro" name="modular_robot">

  <!-- Properties -->
  <xacro:property name="wheel_radius" value="0.05"/>
  <xacro:property name="wheel_width" value="0.02"/>

  <!-- Wheel macro -->
  <xacro:macro name="wheel" params="prefix parent x_offset y_offset">
    <link name="${prefix}_wheel">
      <visual>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
      </visual>
      <collision>
        <geometry>
          <cylinder radius="${wheel_radius}" length="${wheel_width}"/>
        </geometry>
      </collision>
    </link>

    <joint name="${prefix}_wheel_joint" type="continuous">
      <parent link="${parent}"/>
      <child link="${prefix}_wheel"/>
      <origin xyz="${x_offset} ${y_offset} 0" rpy="-1.5708 0 0"/>
      <axis xyz="0 0 1"/>
    </joint>
  </xacro:macro>

  <!-- Base -->
  <link name="base_link">
    <visual>
      <geometry>
        <box size="0.4 0.3 0.1"/>
      </geometry>
    </visual>
  </link>

  <!-- Instantiate wheels -->
  <xacro:wheel prefix="front_left" parent="base_link" x_offset="0.15" y_offset="0.16"/>
  <xacro:wheel prefix="front_right" parent="base_link" x_offset="0.15" y_offset="-0.16"/>
  <xacro:wheel prefix="rear_left" parent="base_link" x_offset="-0.15" y_offset="0.16"/>
  <xacro:wheel prefix="rear_right" parent="base_link" x_offset="-0.15" y_offset="-0.16"/>

</robot>
```
